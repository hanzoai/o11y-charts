- o11y version: '{{ .Values.o11y.image.tag }}'
- otel-collector version: '{{ .Values.otelCollector.image.tag }}'

2. Get the application URL by running these commands:

{{- if .Values.o11y.ingress.enabled -}}
{{- range $host := .Values.o11y.ingress.hosts }}

  {{- range .paths }}
  http{{ if $.Values.o11y.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}

{{- end }}
{{- else if contains "NodePort" .Values.o11y.service.type }}

  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "o11y.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT

{{- else if contains "LoadBalancer" .Values.o11y.service.type }}

     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "o11y.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "o11y.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.o11y.service.port }}

{{- else if contains "ClusterIP" .Values.o11y.service.type }}

  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ .Chart.Name }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component={{ .Values.o11y.name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:{{ .Values.o11y.service.port }} to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME {{ .Values.o11y.service.port }}:{{ .Values.o11y.service.port }}

{{- end }}
